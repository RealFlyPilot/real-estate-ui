## Install

```bash
npm install --save real-estate-ui
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'real-estate-ui'
import 'real-estate-ui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
